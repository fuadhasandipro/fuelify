# 🛢️ Fuel Ration Control System — AI Agent Build Instructions
## Project: Ishwarganj Fuel Pilot PWA
### For: Antigravity Agentic IDE / Any AI Coding Agent

---

## 🎯 PROJECT OVERVIEW

Build a **Progressive Web App (PWA)** using **Next.js 14+ (App Router) + React + Tailwind CSS + Supabase** that allows fuel pump operators in Ishwarganj to:

1. Scan/photograph a vehicle number plate
2. Automatically extract the plate number via OCR (Google Cloud Vision or Tesseract.js)
3. Check a database — if the vehicle fueled up in the last **3 days**, **REJECT** it
4. If allowed, **log the transaction** and auto-generate a **scheduled return slot** exactly 3 days later (in one of three time windows)
5. Work **offline-capable** as a PWA (pump operators may have poor connectivity)

The MP explicitly wants:
- **Simple, fast, crisis-management tool**
- **3-day restriction per vehicle** (motorcycle & car, octane & petrol)
- **Number plate photo → AI reads it → DB check → Allow/Reject**
- **Auto-schedule return slot** in windows: 10AM–1PM / 2PM–6PM / 6PM–10PM
- Pilot deployable **within 3 days**
- Minimal budget

## 🗄️ DATABASE SCHEMA (Supabase / PostgreSQL)

Run this SQL in Supabase SQL Editor:

```sql
-- Pump stations table
CREATE TABLE pump_stations (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  name TEXT NOT NULL,                        -- e.g. "মেসার্স রহিম ফিলিং স্টেশন"
  location TEXT,                             -- area within Ishwarganj
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Operators (pump staff who use the app)
CREATE TABLE operators (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users(id),   -- Supabase auth user
  pump_station_id UUID REFERENCES pump_stations(id),
  full_name TEXT NOT NULL,
  phone TEXT,
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Fuel transaction log (core table)
CREATE TABLE fuel_transactions (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  plate_number TEXT NOT NULL,                -- Normalized plate, e.g. "DHAKA-11-1234"
  plate_image_url TEXT,                      -- Supabase Storage URL of photo
  vehicle_type TEXT CHECK (vehicle_type IN ('motorcycle', 'car', 'other')),
  fuel_type TEXT CHECK (fuel_type IN ('octane', 'petrol', 'diesel')),
  pump_station_id UUID REFERENCES pump_stations(id),
  operator_id UUID REFERENCES operators(id),
  fueled_at TIMESTAMPTZ DEFAULT NOW(),
  next_allowed_at TIMESTAMPTZ,              -- fueled_at + 3 days
  scheduled_slot TEXT,                      -- e.g. "10:00 AM – 1:00 PM"
  scheduled_date DATE,                      -- exact return date
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Index for fast plate lookups
CREATE INDEX idx_fuel_transactions_plate ON fuel_transactions(plate_number);
CREATE INDEX idx_fuel_transactions_fueled_at ON fuel_transactions(fueled_at DESC);

-- Row Level Security
ALTER TABLE fuel_transactions ENABLE ROW LEVEL SECURITY;
ALTER TABLE operators ENABLE ROW LEVEL SECURITY;

-- Operators can only read/write their own pump's records
CREATE POLICY "Operators see own pump records" ON fuel_transactions
  FOR ALL USING (
    pump_station_id = (
      SELECT pump_station_id FROM operators WHERE user_id = auth.uid()
    )
  );

-- Admins can see everything (set role = 'admin' in user metadata)
CREATE POLICY "Admins see all" ON fuel_transactions
  FOR ALL USING (
    (auth.jwt() -> 'user_metadata' ->> 'role') = 'admin'
  );
```

### 1. `/lib/schedule.ts` — Return Slot Scheduler

```typescript
// The MP defined 3 time windows:
// Window A: 10:00 AM – 1:00 PM
// Window B: 2:00 PM – 6:00 PM  
// Window C: 6:00 PM – 10:00 PM

export type TimeSlot = 'A' | 'B' | 'C';

export const TIME_WINDOWS = {
  A: { label: 'সকাল ১০টা – দুপুর ১টা', start: '10:00 AM', end: '1:00 PM', startHour: 10, endHour: 13 },
  B: { label: 'দুপুর ২টা – সন্ধ্যা ৬টা', start: '2:00 PM', end: '6:00 PM', startHour: 14, endHour: 18 },
  C: { label: 'সন্ধ্যা ৬টা – রাত ১০টা', start: '6:00 PM', end: '10:00 PM', startHour: 18, endHour: 22 },
};

export function generateReturnSchedule(fueledAt: Date): {
  returnDate: Date;
  slot: TimeSlot;
  slotLabel: string;
  nextAllowedAt: Date;
} {
  // Add exactly 3 days
  const returnDate = new Date(fueledAt);
  returnDate.setDate(returnDate.getDate() + 3);

  // Determine current time window to assign same window 3 days later
  const currentHour = fueledAt.getHours();
  let slot: TimeSlot;

  if (currentHour >= 10 && currentHour < 13) {
    slot = 'A';
  } else if (currentHour >= 14 && currentHour < 18) {
    slot = 'B';
  } else if (currentHour >= 18 && currentHour < 22) {
    slot = 'C';
  } else {
    // Outside windows — assign next morning window
    slot = 'A';
    // If fueling after 10 PM, schedule is actually 4 days later at morning window
    if (currentHour >= 22) {
      returnDate.setDate(returnDate.getDate() + 1);
    }
  }

  // Set return time to start of the assigned window
  const window = TIME_WINDOWS[slot];
  returnDate.setHours(window.startHour, 0, 0, 0);

  // next_allowed_at = returnDate (the exact return time)
  const nextAllowedAt = new Date(returnDate);

  return {
    returnDate,
    slot,
    slotLabel: window.label,
    nextAllowedAt,
  };
}
```

### 2. `/lib/ocr.ts` — Plate Number Extraction

```typescript
import Tesseract from 'tesseract.js';

export async function extractPlateNumber(imageData: string | File | Blob): Promise<{
  rawText: string;
  plateNumber: string;
  confidence: number;
}> {
  const result = await Tesseract.recognize(imageData, 'eng+ben', {
    // Optimize for license plates
    tessedit_char_whitelist: 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789-',
    tessedit_pageseg_mode: Tesseract.PSM.SINGLE_LINE,
  });

  const rawText = result.data.text.trim();
  const confidence = result.data.confidence;

  // Normalize: remove spaces, uppercase
  const plateNumber = normalizeBangladeshPlate(rawText);

  return { rawText, plateNumber, confidence };
}

function normalizeBangladeshPlate(raw: string): string {
  // Bangladesh plate formats: 
  // Metro: DHAKA-11-1234 or ঢাকা-মেট্রো-ক-১১-১২৩৪
  // District: MYMENSINGH-A-12-1234
  return raw
    .toUpperCase()
    .replace(/\s+/g, '-')
    .replace(/[^A-Z0-9\u0980-\u09FF-]/g, '') // Keep Bengali + Latin + digits + hyphen
    .trim();
}
```

### 3. `/app/api/check-vehicle/route.ts` — Eligibility Check API

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { createSupabaseServerClient } from '@/lib/supabase-server';

const RESTRICTION_DAYS = 3;

export async function POST(req: NextRequest) {
  const { plateNumber } = await req.json();

  if (!plateNumber) {
    return NextResponse.json({ error: 'Plate number required' }, { status: 400 });
  }

  const supabase = createSupabaseServerClient();

  // Find most recent transaction for this plate
  const { data: lastTransaction, error } = await supabase
    .from('fuel_transactions')
    .select('*')
    .eq('plate_number', plateNumber.toUpperCase())
    .order('fueled_at', { ascending: false })
    .limit(1)
    .single();

  if (error && error.code !== 'PGRST116') { // PGRST116 = no rows found
    return NextResponse.json({ error: 'Database error' }, { status: 500 });
  }

  // No previous record — vehicle is allowed
  if (!lastTransaction) {
    return NextResponse.json({
      allowed: true,
      plateNumber,
      lastFueled: null,
      message: 'প্রথমবার — অনুমোদিত',
    });
  }

  const lastFueled = new Date(lastTransaction.fueled_at);
  const now = new Date();
  const daysDiff = (now.getTime() - lastFueled.getTime()) / (1000 * 60 * 60 * 24);

  if (daysDiff < RESTRICTION_DAYS) {
    const nextAllowed = new Date(lastTransaction.next_allowed_at);
    return NextResponse.json({
      allowed: false,
      plateNumber,
      lastFueled: lastTransaction.fueled_at,
      nextAllowedAt: lastTransaction.next_allowed_at,
      scheduledSlot: lastTransaction.scheduled_slot,
      hoursRemaining: Math.ceil((nextAllowed.getTime() - now.getTime()) / (1000 * 60 * 60)),
      message: `অনুমোদিত নয় — ${Math.ceil(RESTRICTION_DAYS - daysDiff)} দিন বাকি`,
    });
  }

  // Restriction period passed — allowed
  return NextResponse.json({
    allowed: true,
    plateNumber,
    lastFueled: lastTransaction.fueled_at,
    message: 'অনুমোদিত',
  });
}
```

### 4. `/app/api/log-fuel/route.ts` — Log Transaction

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { createSupabaseServerClient } from '@/lib/supabase-server';
import { generateReturnSchedule } from '@/lib/schedule';

export async function POST(req: NextRequest) {
  const { plateNumber, plateImageUrl, vehicleType, fuelType, pumpStationId, operatorId } = await req.json();

  const supabase = createSupabaseServerClient();
  const now = new Date();
  const schedule = generateReturnSchedule(now);

  const { data, error } = await supabase
    .from('fuel_transactions')
    .insert({
      plate_number: plateNumber.toUpperCase(),
      plate_image_url: plateImageUrl,
      vehicle_type: vehicleType,
      fuel_type: fuelType,
      pump_station_id: pumpStationId,
      operator_id: operatorId,
      fueled_at: now.toISOString(),
      next_allowed_at: schedule.nextAllowedAt.toISOString(),
      scheduled_slot: schedule.slotLabel,
      scheduled_date: schedule.returnDate.toISOString().split('T')[0],
    })
    .select()
    .single();

  if (error) {
    return NextResponse.json({ error: error.message }, { status: 500 });
  }

  return NextResponse.json({ success: true, transaction: data, schedule });
}
```

## 🔐 MIDDLEWARE (Auth Protection)

```typescript
// middleware.ts
import { createMiddlewareClient } from '@supabase/auth-helpers-nextjs';
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

export async function middleware(req: NextRequest) {
  const res = NextResponse.next();
  const supabase = createMiddlewareClient({ req, res });
  const { data: { session } } = await supabase.auth.getSession();

  const protectedPaths = ['/dashboard', '/scan', '/result', '/admin'];
  const isProtected = protectedPaths.some(p => req.nextUrl.pathname.startsWith(p));

  if (isProtected && !session) {
    return NextResponse.redirect(new URL('/login', req.url));
  }

  // Admin-only routes
  if (req.nextUrl.pathname.startsWith('/admin')) {
    const role = session?.user?.user_metadata?.role;
    if (role !== 'admin') {
      return NextResponse.redirect(new URL('/dashboard', req.url));
    }
  }

  return res;
}

export const config = {
  matcher: ['/dashboard/:path*', '/scan/:path*', '/result/:path*', '/admin/:path*'],
};
```

2. Install all dependencies listed above
3. Set up `.env.local` with Supabase credentials
4. Create Supabase schema (run SQL above)
5. Build in this exact order:
   - `lib/supabase.ts` and `lib/supabase-server.ts`
   - `lib/schedule.ts`
   - `lib/ocr.ts`
   - `lib/plate-validator.ts`
   - `app/api/check-vehicle/route.ts`
   - `app/api/log-fuel/route.ts`
   - `middleware.ts`
   - `app/login/page.tsx`
   - `components/CameraCapture.tsx`
   - `app/scan/page.tsx`
   - `app/result/page.tsx`
   - `app/dashboard/page.tsx`
   - `app/admin/page.tsx`
   - `public/manifest.json`
   - `next.config.js` (with PWA)
6. Test locally on mobile (use `next dev --hostname 0.0.0.0`)
7. Deploy to Vercel
